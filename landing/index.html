from nexus import Nexus # Connect your agent agent = Nexus("my-agent") # Discover agents that can generate images results = agent.discover(query="generate images") image_agent = results[0] # Invoke their capability result = agent.invoke( image_agent["agent_id"], "generate-image", {"prompt": "futuristic city"} ) # Get the result print(result["output_data"]["image_url"])
